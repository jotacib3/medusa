# emitEventStep

Emit an event.

## Example

```ts
import { 
  createWorkflow
} from "@medusajs/workflows-sdk"
import {
  emitEventStep
} from "@medusajs/core-flows"

const helloWorldWorkflow = createWorkflow(
  "hello-world",
  () => {
    emitEventStep({
      eventName: "custom.created",
      data: {
        id: "123"
      }
    })
  }
)
```
